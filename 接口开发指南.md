## 一、环境配置

### 1.1 环境变量文件结构

```
项目根目录/
├── .env                    # 默认环境变量（提交到 git）
├── .env.local              # 本地环境变量（不提交，优先级最高）
├── .env.development        # 开发环境
├── .env.test               # 测试环境
├── .env.production         # 生产环境
└── .env.example            # 环境变量示例（提交到 git）
```

### 1.3 环境变量优先级

```
优先级从高到低：
1. .env.local          （本地覆盖，不提交）
2. .env.[NODE_ENV].local（如 .env.development.local）
3. .env.[NODE_ENV]     （如 .env.development）
4. .env                （默认值）
```

### 1.4 环境变量使用规范

```typescript
// ✅ 正确：服务端使用（所有环境变量）
const dbUrl = process.env.DATABASE_URL;
const jwtSecret = process.env.JWT_SECRET;

// ✅ 正确：客户端使用（必须以 NEXT_PUBLIC_ 开头）
const apiUrl = process.env.NEXT_PUBLIC_API_URL;
const appName = process.env.NEXT_PUBLIC_APP_NAME;

// ❌ 错误：客户端无法访问（没有 NEXT_PUBLIC_ 前缀）
const secret = process.env.JWT_SECRET; // 客户端返回 undefined
```

```prisma
// prisma/schema.prisma

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

// ============================================
// 用户模型
// ============================================
model User {
  id            String    @id @default(cuid())
  email         String    @unique
  username      String    @unique
  password      String
  name          String?
  avatar        String?
  role          Role      @default(USER)
  status        Status    @default(ACTIVE)
  lastLoginAt   DateTime?
  createdAt     DateTime  @default(now())
  updatedAt     DateTime  @updatedAt
  
  // 关联
  satellites    Satellite[]
  positions     Position[]
  logs          OperationLog[]
  
  @@index([email])
  @@index([username])
  @@map("users")
}

enum Role {
  USER
  ADMIN
  SUPER_ADMIN
}

enum Status {
  ACTIVE
  INACTIVE
  BANNED
}

// ============================================
// 卫星模型
// ============================================
model Satellite {
  id              String    @id @default(cuid())
  name            String
  noradId         String    @unique
  internationalId String?   @unique
  country         String?
  operator        String?
  launchDate      DateTime?
  launchSite      String?
  orbitType       OrbitType @default(LEO)
  altitude        Float
  inclination     Float
  period          Float?    // 轨道周期（分钟）
  status          String    @default("active")
  description     String?   @db.Text
  createdAt       DateTime  @default(now())
  updatedAt       DateTime  @updatedAt
  createdById     String?
  
  // 关联
  createdBy       User?     @relation(fields: [createdById], references: [id])
  positions       Position[]
  tleData         TLEData[]
  
  @@index([noradId])
  @@index([orbitType])
  @@index([status])
  @@map("satellites")
}

enum OrbitType {
  LEO   // 低地球轨道
  MEO   // 中地球轨道
  GEO   // 地球同步轨道
  HEO   // 高椭圆轨道
}

// ============================================
// 位置数据模型
// ============================================
model Position {
  id          String   @id @default(cuid())
  satelliteId String
  latitude    Float
  longitude   Float
  altitude    Float
  velocity    Float?   // 速度 km/s
  timestamp   DateTime @default(now())
  source      String   @default("calculated") // calculated, tracked
  
  // 关联
  satellite   Satellite @relation(fields: [satelliteId], references: [id], onDelete: Cascade)

  @@index([timestamp])
  @@index([satelliteId, timestamp])
  @@map("positions")
}

  epoch       DateTime

  createdAt   DateTime @default(now())
  
  // 关联
  satellite   Satellite @relation(fields: [satelliteId], references: [id], onDelete: Cascade)
  
  @@index([satelliteId])
  @@index([epoch])
  @@map("tle_data")
}

// ============================================
// 操作日志模型
// ============================================

  id          String   @id @default(cuid())
```

#### 步骤 2：编写业务逻辑层

```typescript
// lib/services/satellite.service.ts

import { prisma } from '@/lib/db';
import { 
  Satellite, 
  CreateSatelliteRequest, 
  UpdateSatelliteRequest,
  GetSatellitesQuery 
} from '@/types/satellite';
import { Prisma } from '@prisma/client';

// ============================================
// 查询操作
// ============================================

/**
 * 获取卫星列表（分页）
 */
export async function getSatellites(query: GetSatellitesQuery) {
  const {
    page = 1,
    pageSize = 10,
    search,
    orbitType,
    status,
    sortBy = 'createdAt',
    sortOrder = 'desc',
  } = query;

  const skip = (page - 1) * pageSize;

  // 构建查询条件
  const where: Prisma.SatelliteWhereInput = {};

  if (search) {
    where.OR = [
      { name: { contains: search, mode: 'insensitive' } },
      { noradId: { contains: search, mode: 'insensitive' } },
      { country: { contains: search, mode: 'insensitive' } },
    ];
  }

  if (orbitType) {
    where.orbitType = orbitType;
  }

  if (status) {
    where.status = status;
  }

  // 构建排序
  const orderBy: Prisma.SatelliteOrderByWithRelationInput = {
    [sortBy]: sortOrder,
  };

  // 执行查询
  const [list, total] = await Promise.all([
    prisma.satellite.findMany({
      where,
      skip,
      take: pageSize,
      orderBy,
    }),
    prisma.satellite.count({ where }),
  ]);

  return {
    list,
    pagination: {
      page,
      pageSize,
      total,
      totalPages: Math.ceil(total / pageSize),
    },
  };
}

/**
 * 根据 ID 获取卫星
 */
export async function getSatelliteById(id: string) {
  return prisma.satellite.findUnique({
    where: { id },
    include: {
      positions: {
        take: 10,
        orderBy: { timestamp: 'desc' },
      },
    },
  });
}

/**
 * 根据 NORAD ID 获取卫星
 */
export async function getSatelliteByNoradId(noradId: string) {
  return prisma.satellite.findUnique({
    where: { noradId },
  });
}

/**
 * 检查卫星是否存在
 */
export async function satelliteExists(id: string): Promise<boolean> {
  const count = await prisma.satellite.count({
    where: { id },
  });
  return count > 0;
}

// ============================================
// 创建操作
// ============================================

/**
 * 创建卫星
 */
export async function createSatellite(
  data: CreateSatelliteRequest,
  userId?: string
) {
  return prisma.satellite.create({
    data: {
      ...data,
      launchDate: data.launchDate ? new Date(data.launchDate) : undefined,
      createdById: userId,
    },
  });
}

// ============================================
// 更新操作
// ============================================

/**
 * 更新卫星
 */
export async function updateSatellite(
  id: string,
  data: UpdateSatelliteRequest
) {
  return prisma.satellite.update({
    where: { id },
    data: {
      ...data,
      launchDate: data.launchDate ? new Date(data.launchDate) : undefined,
    },
  });
}

/**
 * 更新卫星状态
 */
export async function updateSatelliteStatus(id: string, status: string) {
  return prisma.satellite.update({
    where: { id },
    data: { status },
  });
}

// ============================================
// 删除操作
// ============================================

/**
 * 删除卫星
 */
export async function deleteSatellite(id: string) {
  return prisma.satellite.delete({
    where: { id },
  });
}

/**
 * 批量删除卫星
 */
export async function deleteSatellites(ids: string[]) {
  return prisma.satellite.deleteMany({
    where: {
      id: { in: ids },
    },
  });
}

// ============================================
// 统计操作
// ============================================

/**
 * 获取卫星统计信息
 */
export async function getSatelliteStats() {
  const [total, byOrbitType, byStatus] = await Promise.all([
    prisma.satellite.count(),
    prisma.satellite.groupBy({
      by: ['orbitType'],
      _count: true,
    }),
    prisma.satellite.groupBy({
      by: ['status'],
      _count: true,
    }),
  ]);

  return {
    total,
    byOrbitType: byOrbitType.reduce((acc, item) => {
      acc[item.orbitType] = item._count;
      return acc;
    }, {} as Record<string, number>),
    byStatus: byStatus.reduce((acc, item) => {
      acc[item.status] = item._count;
      return acc;
    }, {} as Record<string, number>),
  };
```

```typescript
/**

 * 获取卫星列表

 * Query Parameters:
 * - page: 页码（默认 1）
 * - pageSize: 每页数量（默认 10）

 * - orbitType: 轨道类型（LEO, MEO, GEO, HEO）

 * - sortBy: 排序字段（name, altitude, createdAt）

export async function GET(request: NextRequest) {
  try {
    const searchParams = request.nextUrl.searchParams;

    // 解析查询参数
    const query: GetSatellitesQuery = {
      page: parseInt(searchParams.get('page') || '1'),
      pageSize: parseInt(searchParams.get('pageSize') || '10'),

      orbitType: searchParams.get('orbitType') as OrbitType || undefined,
      status: searchParams.get('status') || undefined,
      sortBy: searchParams.get('sortBy') as GetSatellitesQuery['sortBy'] || 'createdAt',
      sortOrder: searchParams.get('sortOrder') as 'asc' | 'desc' || 'desc',
    };

    // 参数验证
    if (query.page! < 1) {
      return ResponseUtil.badRequest('页码必须大于 0');
    }

    if (query.pageSize! < 1 || query.pageSize! > 100) {
      return ResponseUtil.badRequest('每页数量必须在 1-100 之间');
    }

    // 获取数据
    const result = await getSatellites(query);

    return ResponseUtil.success(result, '获取卫星列表成功');

    if (!body.altitude || body.altitude <= 0) {
      return ResponseUtil.badRequest('轨道高度必须大于 0');

    if (!body.inclination || body.inclination < 0 || body.inclination > 180) {
```

```typescript
// lib/api/request.ts

import { ApiResponse } from '@/types/response';

/**
 * 基础请求配置
 */
const BASE_URL = process.env.NEXT_PUBLIC_API_URL || '';

/**
 * 通用请求函数
 */
async function request<T>(
  url: string,
  options?: RequestInit
): Promise<ApiResponse<T>> {
  const defaultHeaders: HeadersInit = {
    'Content-Type': 'application/json',
  };

  // 添加认证 token（如果有）
  if (typeof window !== 'undefined') {
    const token = localStorage.getItem('token');
    if (token) {
      defaultHeaders['Authorization'] = `Bearer ${token}`;
    }
  }

  try {
    const response = await fetch(`${BASE_URL}${url}`, {
      ...options,
      headers: {
        ...defaultHeaders,
        ...options?.headers,
      },
    });

    const data: ApiResponse<T> = await response.json();

    // 处理错误响应
    if (data.status === 'error') {
      throw new ApiError(data.message, data.code);
    }

    return data;
  } catch (error) {
    if (error instanceof ApiError) {
      throw error;
    }
    throw new ApiError('网络请求失败', 500);
  }
}

/**
 * 自定义 API 错误类
 */
export class ApiError extends Error {
  constructor(
    message: string,
    public code: number
  ) {
    super(message);
    this.name = 'ApiError';
  }
}

/**
 * GET 请求
 */
export function get<T>(url: string, params?: Record<string, any>) {
  if (params) {
    const searchParams = new URLSearchParams();
    Object.entries(params).forEach(([key, value]) => {
      if (value !== undefined && value !== null) {
        searchParams.append(key, String(value));
      }
    });
    url = `${url}?${searchParams.toString()}`;
  }
  
  return request<T>(url, { method: 'GET' });
}

/**
 * POST 请求
 */
export function post<T>(url: string, body?: any) {
  return request<T>(url, {
    method: 'POST',
    body: JSON.stringify(body),
  });
}

/**
 * PUT 请求
 */
export function put<T>(url: string, body?: any) {
  return request<T>(url, {
    method: 'PUT',
    body: JSON.stringify(body),
  });
}

/**
 * DELETE 请求
 */
export function del<T>(url: string) {
  return request<T>(url, { method: 'DELETE' });
}
```

```typescript
export function withErrorHandler(handler: Function) {
  return async (...args: any[]) => {
    try {
      return await handler(...args);
    } catch (error) {
      console.error('API Error:', error);
      
      // Prisma 错误处理
      if (error instanceof Prisma.PrismaClientKnownRequestError) {
        if (error.code === 'P2002') {
          return ResponseUtil.conflict('数据已存在');
        }
        if (error.code === 'P2025') {
          return ResponseUtil.notFound('资源不存在');
        }
      }
      
      // 验证错误
      if (error instanceof z.ZodError) {
        return ResponseUtil.badRequest(error.errors[0].message);
      }
      
      // 其他错误
      return ResponseUtil.error('服务器内部错误');
    }
  };
}
```

```typescript
// lib/logger.ts

type LogLevel = 'debug' | 'info' | 'warn' | 'error';

class Logger {
  private formatMessage(level: LogLevel, message: string, data?: any) {
    return {
      timestamp: new Date().toISOString(),
      level,
      message,
      data,
    };
  }

  debug(message: string, data?: any) {
    if (process.env.NODE_ENV === 'development') {
      console.debug(this.formatMessage('debug', message, data));
    }
  }

  info(message: string, data?: any) {
    console.info(this.formatMessage('info', message, data));
  }

  warn(message: string, data?: any) {
    console.warn(this.formatMessage('warn', message, data));
  }

  error(message: string, error?: any) {
    console.error(this.formatMessage('error', message, {
      error: error?.message,
      stack: error?.stack,
    }));
  }
}
```
