## 四、后端开发 (API Routes)

### 4.1 API 路由基础

在 `app/api/` 目录下创建 `route.ts` 文件：

```tsx
// app/api/satellites/route.ts
import { NextResponse } from 'next/server';

// GET 请求
export async function GET() {
  const satellites = [
    { id: 1, name: 'ISS', altitude: 408 },
    { id: 2, name: 'Hubble', altitude: 547 },
  ];
  
  return NextResponse.json(satellites);
}

// POST 请求
export async function POST(request: Request) {
  const body = await request.json();
  
  // 处理数据...
  
  return NextResponse.json({ 
    success: true, 
    data: body 
  }, { status: 201 });
}
```

### 4.2 动态路由

```tsx
// app/api/satellites/[id]/route.ts
export async function GET(
  request: Request,
  { params }: { params: { id: string } }
) {
  const id = params.id;
  
  // 查询数据库...
  
  return NextResponse.json({ id, name: 'Satellite ' + id });
}

// PUT 更新
export async function PUT(
  request: Request,
  { params }: { params: { id: string } }
) {
  const id = params.id;
  const body = await request.json();
  
  return NextResponse.json({ id, ...body });
}

// DELETE 删除
export async function DELETE(
  request: Request,
  { params }: { params: { id: string } }
) {
  const id = params.id;
  
  return NextResponse.json({ success: true });
}
```

### 5.3 定义数据模型

```prisma
// prisma/schema.prisma

generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

model Satellite {
  id          String   @id @default(cuid())
  name        String
  noradId     String   @unique
  altitude    Float
  inclination Float
  status      String   @default("active")
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt
  
  // 关联
  positions   Position[]
  
  @@index([noradId])
}

model Position {
  id          String   @id @default(cuid())
  satelliteId String
  latitude    Float
  longitude   Float
  timestamp   DateTime @default(now())
  
  satellite   Satellite @relation(fields: [satelliteId], references: [id])
  
  @@index([satelliteId, timestamp])
}
```

### 5.4 数据库操作

```tsx
// lib/db.ts
import { PrismaClient } from '@prisma/client';

const globalForPrisma = global as unknown as { prisma: PrismaClient };

export const prisma =
  globalForPrisma.prisma ||
  new PrismaClient({
    log: ['query'],
  });

if (process.env.NODE_ENV !== 'production') globalForPrisma.prisma = prisma;
```

### 5.5 CRUD 操作示例

```tsx
// lib/satellite-service.ts
import { prisma } from './db';

// 创建
export async function createSatellite(data: {
  name: string;
  noradId: string;
  altitude: number;
  inclination: number;
}) {
  return prisma.satellite.create({ data });
}

// 查询所有
export async function getAllSatellites() {
  return prisma.satellite.findMany({
    include: { positions: true },
    orderBy: { createdAt: 'desc' },
  });
}

// 查询单个
export async function getSatelliteById(id: string) {
  return prisma.satellite.findUnique({
    where: { id },
    include: { positions: true },
  });
}

// 更新
export async function updateSatellite(id: string, data: Partial<{
  name: string;
  altitude: number;
  status: string;
}>) {
  return prisma.satellite.update({
    where: { id },
    data,
  });
}

// 删除
export async function deleteSatellite(id: string) {
  return prisma.satellite.delete({
    where: { id },
  });
}

// 分页查询
export async function getSatellitesPaginated(page: number, limit: number) {
  const skip = (page - 1) * limit;
  
  const [data, total] = await Promise.all([
    prisma.satellite.findMany({
      skip,
      take: limit,
      orderBy: { createdAt: 'desc' },
    }),
    prisma.satellite.count(),
  ]);
  
  return {
    data,
    pagination: {
      page,
      limit,
      total,
      totalPages: Math.ceil(total / limit),
    },
  };
}
```

### 6.2 Client Components 数据获取

```tsx
'use client';

import { useState, useEffect } from 'react';

interface Satellite {
  id: string;
  name: string;
  altitude: number;
}

export default function SatelliteList() {
  const [satellites, setSatellites] = useState<Satellite[]>([]);
  const [loading, setLoading] = useState(true);
  
  useEffect(() => {
    fetch('/api/satellites')
      .then((res) => res.json())
      .then((data) => {
        setSatellites(data);
        setLoading(false);
      });
  }, []);
  
  if (loading) return <div>加载中...</div>;
  
  return (
    <ul>
      {satellites.map((sat) => (
        <li key={sat.id}>{sat.name}</li>
      ))}
    </ul>
  );
}
```

### 6.3 使用 SWR（推荐）

```bash
pnpm add swr
```

```tsx
// hooks/useSatellites.ts
import useSWR from 'swr';

const fetcher = (url: string) => fetch(url).then((res) => res.json());

export function useSatellites() {
  const { data, error, isLoading, mutate } = useSWR('/api/satellites', fetcher);
  
  return {
    satellites: data,
    isLoading,
    isError: error,
    mutate,
  };
}

// 使用
'use client';

import { useSatellites } from '@/hooks/useSatellites';

export default function SatelliteList() {
  const { satellites, isLoading, isError } = useSatellites();
  
  if (isLoading) return <div>加载中...</div>;
  if (isError) return <div>加载失败</div>;
  
  return (
    <ul>
      {satellites.map((sat: any) => (
        <li key={sat.id}>{sat.name}</li>
      ))}
    </ul>
  );
}
```

## 七、定时任务 (Cron Jobs)

### 7.1 Vercel Cron Jobs

```tsx
// app/api/cron/fetch-satellites/route.ts
import { NextResponse } from 'next/server';

export async function GET(request: Request) {
  // 验证授权
  const authHeader = request.headers.get('authorization');
  if (authHeader !== `Bearer ${process.env.CRON_SECRET}`) {
    return NextResponse.json({ error: 'Unauthorized' }, { status: 401 });
  }
  
  try {
    // 执行定时任务逻辑
    // 例如：从外部 API 获取卫星数据并更新数据库
    
    console.log('Cron job executed at:', new Date().toISOString());
    
    return NextResponse.json({ success: true });
  } catch (error) {
    return NextResponse.json(
      { error: 'Cron job failed' },
      { status: 500 }
    );
  }
}
```
